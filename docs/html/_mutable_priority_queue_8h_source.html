91template <class T>
92void MutablePriorityQueue<T>::heapifyUp(unsigned i) {
93 auto x = H[i];
94 while (i > 1 && *x < *H[parent(i)]) {
95 set(i, H[parent(i)]);
96 i = parent(i);
97 }
98 set(i, x);
99}

106template <class T>
107void MutablePriorityQueue<T>::heapifyDown(unsigned i) {
108 auto x = H[i];
109 while (true) {
110 unsigned k = leftChild(i);
111 if (k >= H.size())
112 break;
113 if (k+1 < H.size() && *H[k+1] < *H[k])
114 ++k; // right child of i
115 if ( ! (*H[k] < *x) )
116 break;
117 set(i, H[k]);
118 i = k;
119 }
120 set(i, x);
121}
