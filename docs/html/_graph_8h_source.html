527inline double haversine(const double lat1, const double lon1, const double lat2, const double lon2)
528{
529 const double earths_radius = 6371000;
530
531 // Get the difference between our two points then convert the difference into radians
532 const double delta_lat = convert_to_radians(lat2 - lat1);
533 const double delta_lon = convert_to_radians(lon2 - lon1);
534
535 const double converted_lat1 = convert_to_radians(lat1);
536 const double converted_lat2 = convert_to_radians(lat2);
537
538 const double aux = pow(sin(delta_lat / 2), 2) + cos(converted_lat1) * cos(converted_lat2) * pow(sin(delta_lon / 2), 2);
539
540 const double c = 2 * atan2(sqrt(aux), sqrt(1 - aux));
541 const double distance = earths_radius * c;
542
543 return distance;
544}
